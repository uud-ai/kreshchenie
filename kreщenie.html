⚙️ Подключение Google Sheets

Шаг 1 — Откройте Google Apps Script

Перейдите по ссылке: script.google.com и нажмите «Новый проект».

Шаг 2 — Вставьте код

Удалите весь текст в редакторе и вставьте этот код:

const SHEET_NAME = 'Крещения'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ 'Дата записи','Номер','Фамилия','Имя','Отчество', 'Дата рождения','Дата крещения','Имя по святцам', 'В честь','Память','Восприемник 1','Восприемница 2', 'Священник','Сан' ]); sheet.getRange(1,1,1,14).setFontWeight('bold'); sheet.setFrozenRows(1); } sheet.appendRow([ data.savedAt, data.nomer, data.familia, data.imya, data.otchestvo, data.dateBorn, data.dateKrest, data.imyaSvyat, data.vChest, data.pamyat, data.vosp1, data.vosp2, data.svyash, data.san ]); return ContentService .createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, error: err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ok: true, status: 'ready'})) .setMimeType(ContentService.MimeType.JSON); }
Шаг 3 — Создайте таблицу

В меню выберите Файл → Создать → Таблицу, дайте ей имя «Книга крещений Усть-Карск», затем вернитесь в редактор скрипта.

Шаг 4 — Привяжите скрипт к таблице

В редакторе скрипта: Расширения → Apps Script — нет, не так. Зайдите в свою таблицу, выберите меню Расширения → Apps Script. Удалите старый код и вставьте код из шага 2 снова.

Шаг 5 — Опубликуйте как веб-приложение

Нажмите кнопку «Развернуть» → «Новое развёртывание».
Тип: Веб-приложение.
Доступ: Все.
Нажмите «Развернуть». Скопируйте URL веб-приложения.
